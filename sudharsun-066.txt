#include<iostream>
using namespace std 
#include<stdio.h>
#include<conio.h>
void main()
{
	char	t;
	int	i,j,l,s,a;
		s=1;
		clrscr();
		t='A';
		a=1;
		cout<<"ENTER THE NUMBER OF ROWS : ";
		cin>>i;
		for(j=1;j<=i;j++)
		{
			if(j%2==0)
			{
				for(l=a;l<=j;l++)
				{
					cout<<t;
					t++;
				}
			}
			if(j%2!=0)
			{
				for(l=a;l<=j;l++)
				{
					cout<<s;
					s++;
				}
			}
			cout<<endl;
		 }
		 getch();
}
